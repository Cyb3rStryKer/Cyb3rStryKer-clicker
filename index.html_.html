<script>


let credits = parseInt(localStorage.getItem('credits')) || 0;


let multiplier = parseFloat(localStorage.getItem('multiplier')) || 1.0;


let inventory = JSON.parse(localStorage.getItem('inventory')) || {};


let xp = parseInt(localStorage.getItem('xp')) || 0;


let level = parseInt(localStorage.getItem('level')) || 1;


let missionProgress = parseInt(localStorage.getItem('missionProgress')) || 0;


let rebirths = parseInt(localStorage.getItem('rebirths')) || 0;

let achievements = JSON.parse(localStorage.getItem('achievements')) || [];


const items = [


{ name: "Pierre", rarity: 0.5, boost: 0, img: "https://img.icons8.com/color/48/rock.png" },


{ name: "Bois", rarity: 0.3, boost: 0.1, img: "https://img.icons8.com/color/48/wood.png" },


{ name: "Argent", rarity: 0.15, boost: 0.2, img: "https://img.icons8.com/color/48/silver-bars.png" },


{ name: "Or", rarity: 0.04, boost: 0.5, img: "https://img.icons8.com/color/48/gold-bars.png" },


{ name: "Diamant", rarity: 0.009, boost: 1.5, img: "https://img.icons8.com/color/48/diamond.png" },


{ name: "Oeuf Mystère (Limité)", rarity: 0.001, boost: 3.0, img: "https://img.icons8.com/color/48/easter-egg.png", limited: true }


];

const achievementList = [


{ id: 'first_click', name: "Premier clic", condition: () => credits >= 1 },


{ id: 'first_box', name: "Première caisse", condition: () => missionProgress >= 1 },


{ id: 'collector', name: "Collectionneur", condition: () => Object.keys(inventory).length >= 5 },


{ id: 'rebirth_master', name: "Reborn !", condition: () => rebirths >= 1 }


];


function switchPage(id) {


document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));


document.querySelectorAll('nav button').forEach(b => b.classList.remove('active'));


document.getElementById(id).classList.add('active');

event.target.classList.add('active');


if (id === 'inventory') updateInventoryDisplay();


if (id === 'achievements') updateAchievements();


}


function clicker() {


let gain = Math.floor(1 * multiplier * (1 + rebirths * 0.5));


credits += gain;


saveGame();


updateDisplay();


}


function openBox() {


if (credits < 50) {


document.getElementById("message").textContent = "Pas assez de crédits !";


return;


}


credits -= 50;


let item = getRandomItem();


if (inventory[item.name]) {


inventory[item.name].count++;


} else {

inventory[item.name] = { count: 1, img: item.img, limited: item.limited || false };


}


multiplier += item.boost;


xp += 10;


missionProgress++;


if (xp >= 100) {


level++;


xp = 0;


}


document.getElementById("message").textContent = `Tu as obtenu : ${item.name} (+${item.boost} boost)`;


saveGame();

updateDisplay();


}


function getRandomItem() {


let r = Math.random();


let sum = 0;


for (let item of items) {


sum += item.rarity;


if (r <= sum) return item;


}


return items[0];

}


function buyBoost(cost, boost) {


if (credits < cost) {


document.getElementById("message").textContent = "Pas assez de crédits pour acheter ce boost.";


return;


}


credits -= cost;


multiplier += boost;


saveGame();


updateDisplay();

}


function rebirth() {


if (level < 10) {


document.getElementById("message").textContent = "Niveau 10 requis pour la renaissance.";


return;


}


rebirths++;


level = 1;


xp = 0;


multiplier = 1.0;

credits = 0;


inventory = {};


missionProgress = 0;


document.getElementById("message").textContent = "Renaissance réussie ! Bonus x0.5 par renaissance.";


saveGame();


updateDisplay();


}


function updateDisplay() {


document.getElementById("credits").textContent = credits;


document.getElementById("multiplier").textContent = multiplier.toFixed(1);

document.getElementById("xp").textContent = xp;


document.getElementById("level").textContent = level;


document.getElementById("rebirths").textContent = rebirths;


document.getElementById("missionProgress").textContent = missionProgress;


document.getElementById("xpBar").style.width = `${xp}%`;


}


function updateInventoryDisplay() {


const list = document.getElementById("inventoryList");


list.innerHTML = "";


for (let name in inventory) {

const data = inventory[name];


list.innerHTML += `<div class="inventory-item ${data.limited ? 'limited' : ''}"><img src="${data.img}" alt="${name}" /><span>${name} x

${data.count}</span></div>`;


}


}


function updateAchievements() {


const list = document.getElementById("achievementsList");


list.innerHTML = "";


achievementList.forEach(a => {


if (a.condition() && !achievements.includes(a.id)) achievements.push(a.id);


if (achievements.includes(a.id)) {

list.innerHTML += `<div class="achievement">

${a.name}</div>`;


}


});


localStorage.setItem('achievements', JSON.stringify(achievements));


}


function saveGame() {


localStorage.setItem('credits', credits);


localStorage.setItem('multiplier', multiplier);


localStorage.setItem('inventory', JSON.stringify(inventory));


localStorage.setItem('xp', xp);


localStorage.setItem('level', level);


localStorage.setItem('missionProgress', missionProgress);


localStorage.setItem('rebirths', rebirths);


localStorage.setItem('achievements', JSON.stringify(achievements));


}


updateDisplay();
